let apiUrl = "http://195.179.229.119/gpt/api.php"
var urlComponents = URLComponents(string: apiUrl)!
urlComponents.queryItems = [
    URLQueryItem(name: "prompt", value: prompt),
    URLQueryItem(name: "api_key", value: apiKey),
    URLQueryItem(name: "model", value: model)
]

let url = urlComponents.url!

// Create the URL request
var request = URLRequest(url: url)
request.httpMethod = "GET"

// Create a URL session
let session = URLSession.shared

// Perform the HTTP GET request
let task = session.dataTask(with: request) { data, response, error in
    if let error = error {
        print("HTTP Error: \(error.localizedDescription)")
        return
    }
    
    guard let httpResponse = response as? HTTPURLResponse, httpResponse.statusCode == 200 else {
        print("HTTP Error: Response status code is not 200")
        return
    }
    
    guard let data = data else {
        print("HTTP Error: No data received")
        return
    }
    
    // Parse JSON response
    do {
        if let json = try JSONSerialization.jsonObject(with: data, options: []) as? [String: Any] {
            print(json)
        } else {
            print("JSON Error: Unable to parse JSON")
        }
    } catch {
        print("JSON Error: \(error.localizedDescription)")
    }
}

// Start the data task
task.resume()

            }
        }

import okhttp3.OkHttpClient
import okhttp3.Request
import okhttp3.Response
import com.google.gson.Gson

fun main() {
    // Define the prompt to be sent
    val prompt = "Please generate a simple blog post according to this title \"What is CHATGPT\""

    // Enter E-mail to generate API
    val apiKey = "Enter your E-mail Address to get the free ChatGPT API"

    // Define the default model if none is specified
    val defaultModel = "gpt-3.5-turbo"

    // Uncomment the model you want to use, and comment out the others
    // val model = "gpt-4"
    // val model = "gpt-4-32k"
    // val model = "gpt-3.5-turbo-0125"
    val model = defaultModel

    // Build the URL to call
    val apiUrl = "http://195.179.229.119/gpt/api.php?prompt=${encodeURIComponent(prompt)}&api_key=${encodeURIComponent(apiKey)}&model=${encodeURIComponent(model)}"

    // Initialize OkHttpClient
    val client = OkHttpClient()

    // Build the request
    val request = Request.Builder()
        .url(apiUrl)
        .build()

    // Execute the request
    try {
        val response: Response = client.newCall(request).execute()
        if (response.isSuccessful) {
            val responseData = response.body?.string()
            val data = Gson().fromJson(responseData, Map::class.java)
            println(data)
        } else {
            println("HTTP Error: ${response.code}")
        }
    } catch (e: Exception) {
        println("Request failed: ${e.message}")
    }
}

// Function to URL-encode a string
fun encodeURIComponent(value: String): String {
    return java.net.URLEncoder.encode(value, "UTF-8")
